Alerter Plungin

Alerter is a plungin which is written in native javascript and framework agnostic. We can use alerter to show and kind of alert or notification. It accepts custom HTML as well.

Alert


	var abc = alerter.alert('My Application', 'This is para 1', 'This is para 2', ['OK', 'Cancel']);
	abc.on('ok', function(){
		alert('I am alert ok');
	});

	abc.on('cancel', function(){
		alert('I am alert cancel');
	});

Warn


	var xyz = alerter.warn('My Application', 'This is new', 'This is old', ['OK', 'Cancel', 'Reload Me']);

	xyz.on('ok', function(){
		alert('I am warn ok');
	});

	xyz.on('cancel', function(){
		alert('I am warn cancel');
	});

	xyz.on('reloadMe', function(){
		alert('I am reload');
	});

Error


	var xyz = alerter.error('My Application', 'Some errors requires your attention', ['OK']);

	xyz.on('ok', function(){
		alert('I am error ok');
	});

Notification


	var xyz = alerter.notification('My Application', 'I will go off in 3 seconds', [3]);
